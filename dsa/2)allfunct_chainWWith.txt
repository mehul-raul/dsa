class HashTable:
    def __init__(self, size):
        self.size = size
        self.table = [[] for _ in range(size)]

    def hash_function(self, key):
        return hash(key) % self.size

    def insert(self, key, value):
        index = self.hash_function(key)
        for pair in self.table[index]:
            if pair[0] == key:
                pair[1] = value
                return
        self.table[index].append([key, value])

    def find(self, key):
        index = self.hash_function(key)
        for pair in self.table[index]:
            if pair[0] == key:
                return pair[1]
        return None

    def delete(self, key):
        index = self.hash_function(key)
        for pair in self.table[index]:
            if pair[0] == key:
                self.table[index].remove(pair)
                return
        raise KeyError("Key not found")

def main():
    size = int(input("Enter the size of the hash table: "))
    dictionary = HashTable(size)

    while True:
        print("\nOperations:")
        print("1. Insert")
        print("2. Find")
        print("3. Delete")
        print("4. Exit")

        choice = int(input("Enter your choice (1-4): "))

        if choice == 1:
            key, value = input("Enter key and value separated by space: ").split()
            dictionary.insert(key, value)
            print("Key-Value pair inserted successfully.")
        elif choice == 2:
            key = input("Enter key to find: ")
            value = dictionary.find(key)
            if value is not None:
                print(f"Value for key {key} is: {value}")
            else:
                print(f"No value found for key {key}.")
        elif choice == 3:
            key = input("Enter key to delete: ")
            try:
                dictionary.delete(key)
                print(f"Key {key} deleted successfully.")
            except KeyError as e:
                print(e)
        elif choice == 4:
            print("Exiting...")
            break
        else:
            print("Invalid choice. Please enter a number between 1 and 4.")

if __name__ == "__main__":
    main()
