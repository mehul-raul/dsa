class TreeNode:
    def __init__(self, data):
        self.data = data
        self.children = []

    def add_child(self, child):
        self.children.append(child)


def print_tree(node, level=0):
    if node is None:
        return

    print("  " * level + node.data)
    for child in node.children:
        print_tree(child, level + 1)


def construct_tree():
    root = TreeNode(input("Enter book name: "))

    while True:
        chapter_name = input("Enter chapter name (or type 'done' to finish): ")
        if chapter_name.lower() == "done":
            break

        chapter = TreeNode(chapter_name)
        root.add_child(chapter)

        while True:
            section_name = input(f"Enter section name for {chapter_name} (or type 'done' to finish): ")
            if section_name.lower() == "done":
                break

            section = TreeNode(section_name)
            chapter.add_child(section)

            while True:
                subsection_name = input(f"Enter subsection name for {section_name} (or type 'done' to finish): ")
                if subsection_name.lower() == "done":
                    break

                subsection = TreeNode(subsection_name)
                section.add_child(subsection)

    return root


def main():
    book_tree = construct_tree()

    print("\nBook Structure:")
    print_tree(book_tree)


if __name__ == "__main__":
    main()
